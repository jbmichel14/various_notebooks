# index_to_id maps the movie index in the data matrix to it's id
index_to_id = {0: first_movie_id}

# method used to create the data matrix
def set_matrix(T,x,current_movie_index):
    #The tag id is the row index (vertical axis)
    tag_id = x[0]
    #The movie id is not the column index (horizontal axis) because not all movies are in scores
    movie_id = x[1]
    value = x[2]
    incr_index = False
    if index_to_id.get(current_movie_index) != movie_id:
        current_movie_index += 1
        index_to_id[current_movie_index] = movie_id
        incr_index = True
    T[x[0]-1][current_movie_index] = value
    return incr_index
